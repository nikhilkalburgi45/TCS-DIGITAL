## Q40 Problem: House Robber

You are given an integer array `nums` where each element represents the amount of money in a house.

You cannot rob two **adjacent houses**.

Return the **maximum amount of money** you can rob.

---

## Example 1

```text
Input: nums = [1,2,3,1]
Output: 4
```

---

## Example 2

```text
Input: nums = [2,7,9,3,1]
Output: 12
```

---

## Example 3

```text
Input: nums = [2,1,1,2]
Output: 4
```

---

### Brute Force Approach (Recursion)

```cpp
#include <bits/stdc++.h>
#include<sstream>
#include<vector>
using namespace std;

// BRUTE FORCE: Try rob or skip (Exponential)
int solve(int i, vector<int> &arr) {

    // Base case: no house left
    if(i < 0) return 0;

    // Option 1: Rob current house
    int rob = arr[i] + solve(i - 2, arr);

    // Option 2: Skip current house
    int skip = solve(i - 1, arr);

    // Take maximum
    return max(rob, skip);
}

int main() {

    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    string line;
    getline(cin, line);

    // Clean input like [1,2,3]
    for (char &ch: line) {
        if (ch == '[' || ch == ']' || ch == ',') {
            ch = ' ';
        }
    }

    vector<int> arr;
    stringstream ss(line);

    int x;
    while (ss >> x) {
        arr.push_back(x);
    }

    int n = arr.size();

    cout << solve(n - 1, arr);
}
```

### Better Approach (Memoization)

```cpp
#include <bits/stdc++.h>
#include<sstream>
#include<vector>
using namespace std;

// BETTER: Top-down DP with memo
int solve(int i, vector<int> &arr, vector<int> &dp) {

    if(i < 0) return 0;

    if(dp[i] != -1) return dp[i];

    int rob = arr[i] + solve(i - 2, arr, dp);
    int skip = solve(i - 1, arr, dp);

    return dp[i] = max(rob, skip);
}

int main() {

    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    string line;
    getline(cin, line);

    for (char &ch: line) {
        if (ch == '[' || ch == ']' || ch == ',') {
            ch = ' ';
        }
    }

    vector<int> arr;
    stringstream ss(line);

    int x;
    while (ss >> x) {
        arr.push_back(x);
    }

    int n = arr.size();

    vector<int> dp(n, -1);

    cout << solve(n - 1, arr, dp);
}
```

### Optimal Approach (Space Optimized DP)

```cpp
#include <bits/stdc++.h>
#include<sstream>
#include<vector>
using namespace std;

// OPTIMAL: Bottom-up with O(1) space
int solve(vector<int> &arr) {

    int n = arr.size();

    if(n == 1) return arr[0];

    int prev2 = arr[0];                // dp[i-2]
    int prev1 = max(arr[0], arr[1]);   // dp[i-1]

    for(int i = 2; i < n; i++){

        int pick = arr[i] + prev2;
        int notPick = prev1;

        int curr = max(pick, notPick);

        prev2 = prev1;
        prev1 = curr;
    }

    return prev1;
}

int main() {

    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    string line;
    getline(cin, line);

    for (char &ch: line) {
        if (ch == '[' || ch == ']' || ch == ',') {
            ch = ' ';
        }
    }

    vector<int> arr;
    stringstream ss(line);

    int x;
    while (ss >> x) {
        arr.push_back(x);
    }

    cout << solve(arr);
}
```
